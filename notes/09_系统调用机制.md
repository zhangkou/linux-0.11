# 第09章：系统调用机制

> 源码文件：`kernel/system_call.s`、`include/linux/sys.h`、`include/unistd.h`
> 核心概念：int 0x80、特权级切换、系统调用表、参数传递

---

## 9.1 概述

系统调用是用户程序进入内核的唯一合法通道。Linux 0.11 使用 **`int 0x80`** 软中断实现系统调用。

**流程概览：**
```
用户程序
  │ int 0x80 (EAX=系统调用号, EBX/ECX/EDX=参数)
  ▼ CPU特权级: Ring3 → Ring0
kernel/system_call.s: _system_call
  │ 保存寄存器 (ds/es/fs/edx/ecx/ebx)
  │ 设置内核段 (ds=es=0x10, fs=0x17)
  │ call sys_call_table[EAX*4]  (跳转对应函数)
  ▼
具体系统调用实现 (如 sys_read, sys_fork...)
  │ 返回值放入 EAX
  ▼
ret_from_sys_call:
  │ 检查信号
  │ 恢复寄存器
  │ iret (Ring0 → Ring3)
  ▼
用户程序继续执行 (EAX = 返回值)
```

---

## 9.2 系统调用门的特殊性

```c
// kernel/sched.c 中：
set_system_gate(0x80, &system_call);

// 与普通中断门的区别：
set_intr_gate(0x20, &timer_interrupt);   // DPL=0，用户态不能直接触发
set_system_gate(0x80, &system_call);     // DPL=3，用户态可以触发
```

**门描述符 DPL 的重要性：**
```
CPU规则：int n 指令只有在当前特权级 CPL ≤ 门描述符DPL 时才允许执行
  - 时钟中断(0x20): DPL=0，用户态(CPL=3)无法主动触发（只能由硬件触发）
  - 系统调用(0x80): DPL=3，用户态可以通过 int 0x80 触发
```

## 9.3 _system_call —— 完整汇编分析

```asm
; kernel/system_call.s 第80~128行

; 进入时栈布局（CPU自动压入的）：
;   SS  (如果发生特权级切换)
;   ESP (用户栈指针)
;   EFLAGS
;   CS  (用户代码段)
;   EIP (用户程序返回地址)

_system_call:
    cmpl $nr_system_calls-1, %eax   ; 检查系统调用号合法性(0~71)
    ja bad_sys_call                   ; 越界则返回-1

    ; 保存寄存器上下文
    push %ds
    push %es
    push %fs
    pushl %edx                        ; 系统调用第3个参数
    pushl %ecx                        ; 系统调用第2个参数
    pushl %ebx                        ; 系统调用第1个参数

    ; 切换到内核数据段
    movl $0x10, %edx
    mov %dx, %ds              ; DS = 内核数据段 (0x10 = GDT[2])
    mov %dx, %es              ; ES = 内核数据段
    movl $0x17, %edx
    mov %dx, %fs              ; FS = 用户数据段 (0x17 = LDT[2])
                               ; FS用于访问用户空间数据（如参数字符串）

    ; 调用对应的系统调用函数
    call _sys_call_table(,%eax,4)
    ; 等价于: call sys_call_table[eax * 4]
    ; 返回值在 EAX 中

    pushl %eax                ; 将返回值压栈保存

    ; 检查当前进程状态，决定是否需要调度
    movl _current, %eax
    cmpl $0, state(%eax)      ; state != RUNNING?
    jne reschedule
    cmpl $0, counter(%eax)    ; counter == 0 (时间片耗尽)?
    je reschedule

ret_from_sys_call:
    ; 检查是否为进程0（进程0不处理信号）
    movl _current, %eax
    cmpl _task, %eax
    je 3f

    ; 检查是否从用户态返回（CS低2位=特权级）
    cmpw $0x0f, CS(%esp)      ; CS = 用户代码段选择子(0x0f)?
    jne 3f
    cmpw $0x17, OLDSS(%esp)   ; SS = 用户数据段选择子(0x17)?
    jne 3f

    ; 检查是否有待处理的信号
    movl signal(%eax), %ebx   ; EBX = 信号位图
    movl blocked(%eax), %ecx
    notl %ecx                  ; ECX = ~blocked (未屏蔽的信号掩码)
    andl %ebx, %ecx            ; ECX = 未屏蔽的待处理信号
    bsfl %ecx, %ecx            ; 找最低位设置的位（最低信号号-1）
    je 3f                      ; 没有信号则跳过
    btrl %ecx, %ebx            ; 清除该信号位
    movl %ebx, signal(%eax)
    incl %ecx                  ; 信号号 = 位号+1
    pushl %ecx
    call _do_signal             ; 处理信号
    popl %eax

3:  ; 恢复寄存器并返回
    popl %eax                  ; 恢复系统调用返回值
    popl %ebx
    popl %ecx
    popl %edx
    pop %fs
    pop %es
    pop %ds
    iret                       ; 中断返回：Ring0 → Ring3
```

**返回时的栈帧（iret前）：**
```
栈地址    内容
ESP+0x00: EAX (系统调用返回值)
ESP+0x04: EBX
ESP+0x08: ECX
ESP+0x0C: EDX
ESP+0x10: FS
ESP+0x14: ES
ESP+0x18: DS
ESP+0x1C: EIP  (用户程序返回地址) ← iret从这里开始弹出
ESP+0x20: CS
ESP+0x24: EFLAGS
ESP+0x28: ESP (用户栈指针)        ← 特权级切换时CPU额外压入
ESP+0x2C: SS
```

## 9.5 用户空间的调用约定

`include/unistd.h` 中定义了调用宏：

```c
// include/unistd.h（节选）

// 无参数系统调用宏
#define _syscall0(type,name) \
type name(void) \
{ \
    long __res; \
    __asm__ volatile ("int $0x80" \
        : "=a" (__res) \
        : "0" (__NR_##name)); \  // EAX = 系统调用号
    if (__res >= 0) \
        return (type) __res; \
    errno = -__res; \
    return -1; \
}

// 一个参数
#define _syscall1(type,name,atype,a) \
type name(atype a) \
{ \
    long __res; \
    __asm__ volatile ("int $0x80" \
        : "=a" (__res) \
        : "0" (__NR_##name), "b" ((long)(a))); \ // EBX = 参数a
    ...
}

// 两个参数 (EBX, ECX)
// 三个参数 (EBX, ECX, EDX)
```

**调用约定总结：**
```
EAX = 系统调用号 (__NR_xxx)
EBX = 第1个参数
ECX = 第2个参数
EDX = 第3个参数
返回值: EAX (>=0成功, <0则 errno=-EAX)
```

## 9.6 FS 寄存器的巧妙用途

系统调用中设置 `FS = 0x17`（用户数据段）的原因：

```c
// include/asm/segment.h 中的访问函数
static inline unsigned char get_fs_byte(const char *addr)
{
    unsigned register char _v;
    __asm__ ("movb %%fs:%1,%0":"=q" (_v):"m" (*addr));
    return _v;
}

static inline void put_fs_long(unsigned long val, unsigned long *addr)
{
    __asm__ ("movl %0,%%fs:%1"::"r" (val),"m" (*addr));
}
```

**为什么需要 FS？**
```
系统调用运行在内核（Ring 0），DS=ES=0x10（内核数据段）
但参数可能是用户空间的指针（如文件名字符串）
此时不能直接用 DS 访问用户空间（因为基址不同）

解决：用 FS 段寄存器（指向用户数据段 0x17）专门访问用户空间数据
  get_fs_byte(ptr):  从 FS:ptr 读取 = 从用户空间读取
  put_fs_long(v,p):  写入 FS:p    = 写入用户空间
```

## 9.8 系统调用的完整时序

```
用户程序
  open("/etc/hosts", O_RDONLY, 0)
    │
    ▼ (lib/open.c)
  EAX=5(__NR_open), EBX=路径, ECX=标志, EDX=模式
  int 0x80
    │
    ▼ CPU特权级切换 Ring3→Ring0
    │ CPU自动压栈: SS,ESP,EFLAGS,CS,EIP
  _system_call:
    │ 检查EAX=5合法
    │ push ds/es/fs/edx/ecx/ebx
    │ ds=es=0x10, fs=0x17
    │ call sys_call_table[5*4]
    │
    ▼
  sys_open(filename, flag, mode)   (fs/open.c)
    │ get_fs_byte()读取文件名（通过FS访问用户空间）
    │ 分配文件描述符
    │ 打开文件...
    │ return fd (文件描述符，如 3)
    │
    ▼ 返回 EAX=3
  ret_from_sys_call:
    │ 检查信号
    │ pop ebx/ecx/edx/fs/es/ds
    │ iret
    │
    ▼ Ring0→Ring3
用户程序
  open() 返回 3
```

## 9.9 总结

| 方面 | 实现 |
|------|------|
| 触发方式 | `int 0x80`（软中断） |
| 调用号传递 | EAX |
| 参数传递 | EBX（1）, ECX（2）, EDX（3） |
| 返回值 | EAX（≥0成功，<0是负的errno） |
| 特权切换 | CPU自动：Ring3→Ring0→Ring3 |
| 用户空间访问 | FS寄存器 + get_fs_byte/put_fs_long |
| 最大调用号 | 71（共72个系统调用） |
| 信号处理时机 | 系统调用返回时（ret_from_sys_call） |
