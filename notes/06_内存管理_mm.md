# 第06章：内存管理 mm/memory.c

> 源码文件：`mm/memory.c`、`mm/page.s`
> 核心概念：物理页分配、请求调页、写时复制（COW）

---

## 6.1 概述

Linux 0.11 的内存管理实现了三个关键机制：
1. **物理页分配/释放** — mem_map[] 位图跟踪每个物理页的引用计数
2. **请求调页（Demand Paging）** — 进程访问未映射页时，通过缺页异常按需分配
3. **写时复制（Copy-On-Write, COW）** — fork() 时不复制物理页，写操作触发时才复制

---

## 6.2 关键常量与数据结构

```c
// mm/memory.c 第43~57行
#define LOW_MEM      0x100000        // 1MB：内核占用区，不纳入分页管理
#define PAGING_MEMORY (15*1024*1024) // 15MB：可分页的物理内存（1MB~16MB）
#define PAGING_PAGES  (PAGING_MEMORY>>12) // 15MB/4KB = 3840 个物理页
#define MAP_NR(addr)  (((addr)-LOW_MEM)>>12) // 物理地址→mem_map下标
#define USED 100    // 初始化时标记内核区和缓冲区为"已使用"

// 物理页引用计数数组（每字节对应一个4KB物理页）
static unsigned char mem_map[PAGING_PAGES] = {0,};
// 下标0 对应 物理地址 0x100000（1MB）
// 下标i 对应 物理地址 LOW_MEM + i*4096
// 值 0 = 空闲, 1 = 被一个进程使用, >1 = 被多个进程共享（COW）
```

---

## 6.3 内存布局（运行时）

```
物理地址
0x000000 ┌──────────────────────┐
         │  内核代码/数据/页表   │  (约几百KB)
         │  缓冲区               │  (1~4MB，由buffer_init管理)
0x100000 ├──────────────────────┤  ← LOW_MEM (1MB)
         │                      │
         │  可分页内存           │  由 mem_map[] 管理
         │  (15MB)              │  PAGING_PAGES = 3840 个页
         │                      │
0xFFFFFF └──────────────────────┘  ← HIGH_MEMORY (= memory_end, 最大16MB)
```

## 6.4 mem_init() —— 初始化物理页分配器

```c
// mm/memory.c 第399~411行
void mem_init(long start_mem, long end_mem)
{
    int i;

    HIGH_MEMORY = end_mem;       // 记录内存上限

    // 第一步：将所有页标记为 USED=100（已使用）
    for (i=0 ; i<PAGING_PAGES ; i++)
        mem_map[i] = USED;

    // 第二步：将主内存区（start_mem ~ end_mem）标记为空闲
    i = MAP_NR(start_mem);       // 主内存起始页的下标
    end_mem -= start_mem;
    end_mem >>= 12;              // 主内存的页数
    while (end_mem-- > 0)
        mem_map[i++] = 0;        // 清零 = 空闲
}
```

**效果示意（8MB内存，主内存从2MB开始）：**
```
mem_map[0..255]   = 100  (物理1MB~2MB = 缓冲区，不可分页)
mem_map[256..1791] = 0   (物理2MB~8MB = 主内存，可用)
mem_map[1792..3839] = 100 (物理8MB~16MB = 超出实际内存，不可用)
```

## 6.5 get_free_page() —— 分配一个物理页

这是内核最重要的内存分配函数。它用内联汇编实现，并发性能更好：

```c
// mm/memory.c 第63~83行
unsigned long get_free_page(void)
{
    register unsigned long __res asm("ax");

    __asm__(
        "std ; repne ; scasb\n\t"  // 从高地址向低地址扫描 mem_map[]
                                    // 找第一个值为0（空闲）的字节
                                    // std=方向递减, repne=不等则继续, scasb=比较AL与ES:EDI
        "jne 1f\n\t"               // 没找到则跳转标号1（返回0）
        "movb $1,1(%%edi)\n\t"     // 找到了：将该页标记为1（已使用）
                                    // edi+1 是因为scasb会先递减edi
        "sall $12,%%ecx\n\t"       // ecx = 页号 × 4096 = 页内偏移（相对LOW_MEM）
        "addl %2,%%ecx\n\t"        // ecx += LOW_MEM → 得到物理地址
        "movl %%ecx,%%edx\n\t"     // edx = 物理地址
        "movl $1024,%%ecx\n\t"     // 要清零1024个long = 4KB
        "leal 4092(%%edx),%%edi\n\t" // edi指向页末尾
        "rep ; stosl\n\t"          // 将整页清零（防止信息泄露）
        "movl %%edx,%%eax\n"       // 返回物理地址
        "1:"
        : "=a" (__res)
        : "0" (0), "i" (LOW_MEM), "c" (PAGING_PAGES),
          "D" (mem_map+PAGING_PAGES-1)  // 从数组末尾开始扫描
        : "di","cx","dx"
    );
    return __res;  // 返回物理地址，失败返回0
}
```

**关键设计：**
- 从高地址向低扫描，优先使用高地址物理页
- 分配时**立即清零**整页（安全性考虑）
- 返回**物理地址**（不是虚拟地址）

## 6.6 free_page() —— 释放一个物理页

```c
// mm/memory.c 第89~99行
void free_page(unsigned long addr)
{
    if (addr < LOW_MEM) return;           // 内核区不管
    if (addr >= HIGH_MEMORY)
        panic("trying to free nonexistent page");
    addr -= LOW_MEM;
    addr >>= 12;                          // 转换为 mem_map 下标
    if (mem_map[addr]--) return;          // 引用计数减1，如果还>0则返回
    mem_map[addr] = 0;
    panic("trying to free free page");    // 引用计数已经为0，double free！
}
```

## 6.7 copy_page_tables() —— fork 时复制页表（COW核心）

这是写时复制的实现核心，在 fork() 时调用：

```c
// mm/memory.c 第150~189行
int copy_page_tables(unsigned long from, unsigned long to, long size)
{
    unsigned long *from_page_table, *to_page_table;
    unsigned long this_page;
    unsigned long *from_dir, *to_dir;
    unsigned long nr;

    // 要求4MB对齐（一个页目录项覆盖4MB）
    from_dir = (unsigned long *)((from>>20) & 0xffc); // 页目录项地址
    to_dir   = (unsigned long *)((to>>20)   & 0xffc);
    size = ((unsigned)(size+0x3fffff)) >> 22; // 转换为页目录项数

    for (; size-->0; from_dir++, to_dir++) {
        if (!(1 & *from_dir)) continue;    // 源页目录项无效则跳过
        from_page_table = (unsigned long *)(0xfffff000 & *from_dir);

        // 为子进程分配新页表
        if (!(to_page_table = (unsigned long *)get_free_page()))
            return -1;
        *to_dir = ((unsigned long)to_page_table) | 7; // 设置子进程页目录项

        // 特殊情况：from==0 是进程0→进程1 的首次fork
        // 只复制160页(640KB)，避免浪费（低1MB不做COW）
        nr = (from == 0) ? 0xA0 : 1024;

        for (; nr-- > 0; from_page_table++, to_page_table++) {
            this_page = *from_page_table;
            if (!(1 & this_page)) continue;  // 页不存在则跳过

            this_page &= ~2;                  // ★关键：清除R/W位，设为只读
            *to_page_table = this_page;       // 子进程页表项 = 只读

            if (this_page > LOW_MEM) {
                *from_page_table = this_page; // ★父进程页也设为只读
                this_page -= LOW_MEM;
                this_page >>= 12;
                mem_map[this_page]++;         // 引用计数+1（共享）
            }
        }
    }
    invalidate(); // 刷新TLB（令CPU重新加载页表）
    return 0;
}
```

**COW机制图示：**
```
fork() 后：
父进程页表                子进程页表
PTE → 物理页A(只读)  ←→  PTE → 物理页A(只读)
              mem_map[A] = 2 (共享)

父进程写入物理页A时，触发缺页异常（写保护）：
→ do_wp_page() 分配新页B，复制A→B，父进程改用B
→ mem_map[A]--,  mem_map[B] = 1
→ 父进程继续，子进程仍用A
```

## 6.8 do_wp_page() —— 处理写保护异常

当进程试图写一个只读页时，CPU触发缺页异常，最终调用此函数：

```c
// mm/memory.c 第247~258行
void do_wp_page(unsigned long error_code, unsigned long address)
{
    // 根据线性地址找到对应的页表项
    un_wp_page((unsigned long *)
        (((address>>10) & 0xffc) +      // 页表内偏移
         (0xfffff000 &                   // 页表物理基地址
          *((unsigned long *)((address>>20) & 0xffc)))));
}

void un_wp_page(unsigned long *table_entry)
{
    unsigned long old_page, new_page;

    old_page = 0xfffff000 & *table_entry;  // 当前物理页地址

    // 情况1：该页只有一个引用（引用计数=1），直接恢复可写
    if (old_page >= LOW_MEM && mem_map[MAP_NR(old_page)] == 1) {
        *table_entry |= 2;    // 设置R/W=1（可写）
        invalidate();
        return;
    }

    // 情况2：该页被多个进程共享（引用计数>1），需要复制
    if (!(new_page = get_free_page()))
        oom();
    if (old_page >= LOW_MEM)
        mem_map[MAP_NR(old_page)]--;  // 减少原页引用计数
    *table_entry = new_page | 7;      // 指向新页，设为可写
    invalidate();
    copy_page(old_page, new_page);    // 将原页内容复制到新页
}
```

## 6.9 do_no_page() —— 处理缺页异常（请求调页）

当进程访问一个完全未映射的地址时触发：

```c
// mm/memory.c 第365~397行
void do_no_page(unsigned long error_code, unsigned long address)
{
    int nr[4];
    unsigned long tmp;
    unsigned long page;
    int block, i;

    address &= 0xfffff000;           // 对齐到页边界
    tmp = address - current->start_code;  // 相对于代码段起始的偏移

    // 情况1：地址超出可执行文件范围（如栈、堆扩展）
    if (!current->executable || tmp >= current->end_data) {
        get_empty_page(address);     // 分配一个清零的新页
        return;
    }

    // 情况2：尝试共享其他进程的同一可执行文件的页（节省内存）
    if (share_page(tmp))
        return;

    // 情况3：从磁盘上的可执行文件加载对应的块
    if (!(page = get_free_page()))
        oom();

    // 计算该地址对应可执行文件的块号（一个4KB页跨4个1KB块）
    block = 1 + tmp/BLOCK_SIZE;   // +1 是因为第0块是a.out文件头
    for (i=0; i<4; block++, i++)
        nr[i] = bmap(current->executable, block);  // 获取文件块号

    bread_page(page, current->executable->i_dev, nr); // 从磁盘读入

    // 清零超出 end_data 的部分（BSS段为全零）
    i = tmp + 4096 - current->end_data;
    tmp = page + 4096;
    while (i-- > 0) {
        tmp--;
        *(char *)tmp = 0;
    }

    if (put_page(page, address))   // 建立线性地址→物理地址的映射
        return;
    free_page(page);
    oom();
}
```

## 6.10 缺页异常的完整路径

```
进程访问未映射的线性地址
    │
    ▼
CPU 产生 Page Fault (#14 异常)
    │
    ▼
mm/page.s: _page_fault (汇编入口)
    │ 读取 CR2 (产生缺页的线性地址)
    │ 读取 error_code
    ▼
mm/memory.c: do_page_fault()
    │
    ├── error_code & 1 == 1 (页存在但写保护)
    │       └──▶ do_wp_page()  → COW复制
    │
    └── error_code & 1 == 0 (页不存在)
            └──▶ do_no_page() → 请求调页
```

**error_code 位含义：**
```
bit0 = P  (0=页不存在, 1=写保护违规)
bit1 = R/W (0=读操作触发, 1=写操作触发)
bit2 = U/S (0=内核模式, 1=用户模式)
```

## 6.11 地址空间布局（每个进程）

每个进程有独立的 64MB 虚拟地址空间（由 LDT 决定）：

```
进程n的虚拟地址 = n × 64MB (线性地址)
    │
    ▼ LDT 基址映射
进程n的线性地址:
  n×64MB + 0x00000  代码段开始
  n×64MB + end_code 代码段结束
  n×64MB + end_data 数据段结束
  n×64MB + brk      堆顶
  n×64MB + 64MB-4   栈顶
```

**进程地址空间对应关系：**
```
进程1: 线性地址 0x04000000 ~ 0x07FFFFFF  (64MB)
进程2: 线性地址 0x08000000 ~ 0x0BFFFFFF  (64MB)
进程3: 线性地址 0x0C000000 ~ 0x0FFFFFFF  (64MB)
...
最多 64 个进程 × 64MB = 4096MB (超过了32位地址空间)
但实际物理内存上限16MB，通过页表实现共享
```

## 6.12 总结

| 函数 | 作用 |
|------|------|
| `mem_init()` | 初始化 mem_map[] 位图 |
| `get_free_page()` | 分配一个4KB物理页（清零后返回物理地址） |
| `free_page()` | 释放一个物理页（减引用计数） |
| `copy_page_tables()` | fork时复制页表（设为只读，实现COW） |
| `free_page_tables()` | exit时释放进程所有页 |
| `do_wp_page()` | COW触发时复制物理页 |
| `do_no_page()` | 缺页时从磁盘加载 |
| `put_page()` | 建立线性地址→物理页的映射 |
