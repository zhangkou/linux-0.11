# 第08章：进程创建与退出

> 源码文件：`kernel/fork.c`、`kernel/exit.c`
> 核心概念：fork()的完整实现、COW页表复制、进程退出与僵尸态、wait()

---

## 8.1 fork() 的调用链

```
用户程序调用 fork()
    │
    ▼ int 0x80 (系统调用)
kernel/system_call.s: _sys_fork
    │ call _find_empty_process  (找空槽)
    │ push 寄存器参数
    │ call _copy_process        (核心)
    ▼
kernel/fork.c: copy_process()
    │ get_free_page()           (分配task_struct+内核栈)
    │ 复制task_struct
    │ 设置TSS寄存器值
    │ copy_mem()                (复制地址空间)
    │   └── copy_page_tables()  (COW页表复制)
    │ 增加文件/inode引用计数
    │ 设置GDT中的TSS/LDT描述符
    └── 设置 state = TASK_RUNNING
```

## 8.2 find_empty_process() —— 分配PID和进程槽

```c
// kernel/fork.c 第135~147行
int find_empty_process(void)
{
    int i;

    repeat:
        if ((++last_pid) < 0) last_pid = 1;  // PID溢出则回绕到1
        // 检查新PID是否已被使用
        for (i=0; i<NR_TASKS; i++)
            if (task[i] && task[i]->pid == last_pid)
                goto repeat;

    // 找空的进程槽（task[0]是进程0，永不为空）
    for (i=1; i<NR_TASKS; i++)
        if (!task[i])
            return i;   // 返回空槽下标

    return -EAGAIN;  // 进程数已满（64个）
}
```

## 8.3 _sys_fork —— 汇编入口（传递所有寄存器）

```asm
; kernel/system_call.s 第207~219行
_sys_fork:
    call _find_empty_process   ; 找空进程槽，返回值在EAX
    testl %eax, %eax
    js 1f                      ; EAX < 0 说明失败，直接返回错误码

    ; 将所有寄存器作为参数压栈（传给copy_process）
    push %gs
    pushl %esi
    pushl %edi
    pushl %ebp
    pushl %eax      ; nr (进程槽下标)
    call _copy_process
    addl $20, %esp  ; 清除5个参数（20字节）
1:  ret
```

**copy_process 的参数来自两处：**
1. `_sys_fork` 压栈的：gs, esi, edi, ebp, nr
2. `_system_call` 之前已经压栈的：ebx, ecx, edx, fs, es, ds, eip, cs, eflags, esp(用户栈), ss

## 8.4 copy_process() —— 进程复制核心

```c
// kernel/fork.c 第68~133行
int copy_process(int nr, long ebp, long edi, long esi, long gs, long none,
    long ebx, long ecx, long edx,
    long fs, long es, long ds,
    long eip, long cs, long eflags, long esp, long ss)
{
    struct task_struct *p;
    int i;
    struct file *f;

    // ===== 第1步：分配进程描述符页 =====
    p = (struct task_struct *) get_free_page();
    if (!p) return -EAGAIN;
    task[nr] = p;

    // ===== 第2步：复制父进程的task_struct =====
    *p = *current;   // 整体复制（注意：没有复制内核栈！）
    p->state = TASK_UNINTERRUPTIBLE;  // 暂时设为不可中断，初始化完成前不被调度

    // ===== 第3步：修改子进程特有的字段 =====
    p->pid = last_pid;
    p->father = current->pid;
    p->counter = p->priority;  // 重置时间片
    p->signal = 0;             // 清除信号
    p->alarm = 0;              // 清除定时器
    p->leader = 0;             // 不继承会话leader
    p->utime = p->stime = 0;   // 清除时间统计
    p->cutime = p->cstime = 0;
    p->start_time = jiffies;

    // ===== 第4步：设置子进程的 TSS（保存上下文） =====
    p->tss.back_link = 0;
    p->tss.esp0 = PAGE_SIZE + (long)p;  // 内核栈顶 = 页末尾
    p->tss.ss0 = 0x10;                  // 内核数据段

    // 子进程将从 fork() 系统调用的 iret 处返回
    p->tss.eip = eip;        // 返回地址（用户态fork()调用后的下一条指令）
    p->tss.eflags = eflags;
    p->tss.eax = 0;          // ★关键：子进程 fork() 返回值为0！
    p->tss.ecx = ecx;
    p->tss.edx = edx;
    p->tss.ebx = ebx;
    p->tss.esp = esp;        // 用户栈指针（与父进程相同，因COW共享）
    p->tss.ebp = ebp;
    p->tss.esi = esi;
    p->tss.edi = edi;
    p->tss.es = es & 0xffff;
    p->tss.cs = cs & 0xffff;
    p->tss.ss = ss & 0xffff;
    p->tss.ds = ds & 0xffff;
    p->tss.fs = fs & 0xffff;
    p->tss.gs = gs & 0xffff;
    p->tss.ldt = _LDT(nr);   // 子进程自己的LDT选择子
    p->tss.trace_bitmap = 0x80000000;

    // 保存父进程FPU状态给子进程
    if (last_task_used_math == current)
        __asm__("clts ; fnsave %0" :: "m" (p->tss.i387));

    // ===== 第5步：复制内存空间（COW） =====
    if (copy_mem(nr, p)) {   // 失败则回滚
        task[nr] = NULL;
        free_page((long)p);
        return -EAGAIN;
    }

    // ===== 第6步：增加共享资源的引用计数 =====
    for (i=0; i<NR_OPEN; i++)
        if (f = p->filp[i])
            f->f_count++;        // 每个打开的文件引用计数+1
    if (current->pwd)
        current->pwd->i_count++; // 当前目录inode引用+1
    if (current->root)
        current->root->i_count++;
    if (current->executable)
        current->executable->i_count++;

    // ===== 第7步：在GDT中注册子进程的TSS/LDT =====
    set_tss_desc(gdt+(nr<<1)+FIRST_TSS_ENTRY, &(p->tss));
    set_ldt_desc(gdt+(nr<<1)+FIRST_LDT_ENTRY, &(p->ldt));

    // ===== 第8步：设为就绪态 =====
    p->state = TASK_RUNNING;  // 子进程现在可以被调度

    return last_pid;  // 父进程返回子进程PID
}
```

**fork() 返回值的秘密：**
```
父进程：copy_process() 返回 last_pid → 系统调用返回给用户 = 子进程PID
子进程：p->tss.eax = 0 → 子进程被调度时从TSS恢复EAX=0 → fork()返回0
```

## 8.5 copy_mem() —— 地址空间布局设置

```c
// kernel/fork.c 第39~61行
int copy_mem(int nr, struct task_struct *p)
{
    unsigned long old_data_base, new_data_base, data_limit;
    unsigned long old_code_base, new_code_base, code_limit;

    code_limit = get_limit(0x0f);     // 从LDT读取代码段限长（0x0f = 用户代码段选择子）
    data_limit = get_limit(0x17);     // 数据段限长（0x17 = 用户数据段选择子）
    old_code_base = get_base(current->ldt[1]);  // 父进程代码段基址
    old_data_base = get_base(current->ldt[2]);  // 父进程数据段基址

    // Linux 0.11 要求代码段和数据段基址相同（flat model）
    if (old_data_base != old_code_base)
        panic("We don't support separate I&D");

    // 子进程的地址空间基址 = nr × 64MB
    new_data_base = new_code_base = nr * 0x4000000;
    p->start_code = new_code_base;

    // 修改子进程LDT：设置新的段基址
    set_base(p->ldt[1], new_code_base);
    set_base(p->ldt[2], new_data_base);

    // 复制页表（COW）：将父进程的页表复制给子进程
    if (copy_page_tables(old_data_base, new_data_base, data_limit)) {
        free_page_tables(new_data_base, data_limit);
        return -ENOMEM;
    }
    return 0;
}
```

**地址空间分配示意（进程号与基址的关系）：**
```
进程0: 基址 0×64MB = 0x00000000   (内核进程，特殊)
进程1: 基址 1×64MB = 0x04000000
进程2: 基址 2×64MB = 0x08000000
...
进程n: 基址 n×64MB = n×0x4000000

每个进程的线性地址空间：[n×64MB, (n+1)×64MB)
```

## 8.6 do_exit() —— 进程退出

```c
// kernel/exit.c 第102~135行
int do_exit(long code)
{
    int i;

    // ===== 第1步：释放进程的所有页表 =====
    free_page_tables(get_base(current->ldt[1]), get_limit(0x0f));  // 代码段
    free_page_tables(get_base(current->ldt[2]), get_limit(0x17));  // 数据段

    // ===== 第2步：处理子进程（过继给进程1） =====
    for (i=0; i<NR_TASKS; i++)
        if (task[i] && task[i]->father == current->pid) {
            task[i]->father = 1;   // 将孤儿进程过继给init（进程1）
            // 如果子进程已经是僵尸，通知init处理
            if (task[i]->state == TASK_ZOMBIE)
                (void) send_sig(SIGCHLD, task[1], 1);
        }

    // ===== 第3步：关闭所有打开的文件 =====
    for (i=0; i<NR_OPEN; i++)
        if (current->filp[i])
            sys_close(i);

    // ===== 第4步：释放inode引用 =====
    iput(current->pwd);
    current->pwd = NULL;
    iput(current->root);
    current->root = NULL;
    iput(current->executable);
    current->executable = NULL;

    // ===== 第5步：终端处理 =====
    if (current->leader && current->tty >= 0)
        tty_table[current->tty].pgrp = 0;  // 释放终端控制

    // ===== 第6步：FPU处理 =====
    if (last_task_used_math == current)
        last_task_used_math = NULL;

    // ===== 第7步：会话结束处理 =====
    if (current->leader)
        kill_session();  // 向同会话进程发SIGHUP

    // ===== 第8步：变成僵尸，通知父进程 =====
    current->state = TASK_ZOMBIE;
    current->exit_code = code;
    tell_father(current->father);  // 向父进程发 SIGCHLD
    schedule();  // 让出CPU，永不返回（除非schedule()调用do_exit？不会）
    return -1;   // 抑制编译器警告，实际不会执行到这里
}
```

**僵尸进程（Zombie）的意义：**
```
进程退出后不立即释放task_struct，而是变成僵尸态。
保留task_struct是为了让父进程能通过wait()获取退出码。
父进程调用wait()后，才真正释放（release()）。

如果父进程先退出：
  子进程的father = 1（过继给init）
  init会定期调用wait()，收割孤儿进程的僵尸
```

## 8.7 sys_waitpid() —— 等待子进程

```c
// kernel/exit.c 第142~195行
int sys_waitpid(pid_t pid, unsigned long *stat_addr, int options)
{
    int flag, code;
    struct task_struct **p;

    verify_area(stat_addr, 4);  // 验证用户地址可写
repeat:
    flag = 0;
    // 遍历所有进程，找匹配的子进程
    for (p = &LAST_TASK; p > &FIRST_TASK; --p) {
        if (!*p || *p == current) continue;
        if ((*p)->father != current->pid) continue;  // 不是自己的子进程

        // pid参数的不同含义：
        // pid > 0: 等待指定PID的子进程
        // pid = 0: 等待同进程组的任意子进程
        // pid =-1: 等待任意子进程
        // pid <-1: 等待进程组号为|pid|的子进程
        if (pid > 0 && (*p)->pid != pid) continue;
        if (!pid && (*p)->pgrp != current->pgrp) continue;
        if (pid != -1 && (*p)->pgrp != -pid) continue;

        switch ((*p)->state) {
            case TASK_STOPPED:           // 子进程已停止
                if (!(options & WUNTRACED)) continue;
                put_fs_long(0x7f, stat_addr);
                return (*p)->pid;

            case TASK_ZOMBIE:            // 子进程已退出（僵尸）
                current->cutime += (*p)->utime;  // 累计子进程时间
                current->cstime += (*p)->stime;
                flag = (*p)->pid;
                code = (*p)->exit_code;
                release(*p);             // ★真正释放子进程资源
                put_fs_long(code, stat_addr);  // 将退出码写到用户空间
                return flag;             // 返回子进程PID

            default:                     // 子进程还在运行
                flag = 1;
                continue;
        }
    }
    if (flag) {             // 找到了子进程但尚未退出
        if (options & WNOHANG) return 0;  // 非阻塞则立即返回
        current->state = TASK_INTERRUPTIBLE;
        schedule();         // 睡眠等待
        if (!(current->signal &= ~(1<<(SIGCHLD-1))))
            goto repeat;    // 被SIGCHLD唤醒，再次检查
        else
            return -EINTR;  // 被其他信号打断
    }
    return -ECHILD;         // 没有符合条件的子进程
}
```

## 8.8 进程生命周期完整图

```
fork()
  │
  ▼
TASK_UNINTERRUPTIBLE (初始化中)
  │
  │ copy_process() 完成，state = RUNNING
  ▼
TASK_RUNNING ←──────────────────┐
  │                             │
  │ 时间片耗尽/主动让出          │
  │ schedule()                  │
  ▼                             │
等待CPU...                      │ 被调度/被唤醒
  │                             │
  │ IO/资源等待                  │
  ▼                             │
TASK_INTERRUPTIBLE ─────────────┘
TASK_UNINTERRUPTIBLE ───────────┘
  │
  │ exit()/do_exit()
  ▼
TASK_ZOMBIE
  │
  │ 父进程调用 wait()/waitpid()
  │ release() 释放 task_struct
  ▼
（进程彻底消失）
```

## 8.9 总结

| 操作 | 函数 | 关键动作 |
|------|------|---------|
| 创建进程 | `copy_process()` | 分配页、复制task_struct、设置TSS、COW页表 |
| 子进程返回值 | TSS中 `eax=0` | 子进程被调度时EAX=0，即fork()返回0 |
| 地址空间分配 | `copy_mem()` | 子进程基址 = nr×64MB，LDT中设置 |
| 退出 | `do_exit()` | 释放页表、关文件、变僵尸、通知父进程 |
| 孤儿进程 | `do_exit()` | 过继给进程1（init） |
| 收割僵尸 | `sys_waitpid()` | 获取退出码，调用release()彻底释放 |
