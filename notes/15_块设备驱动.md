# 第15章：块设备驱动

> 源码文件：`kernel/blk_drv/ll_rw_blk.c`、`kernel/blk_drv/hd.c`、`kernel/blk_drv/blk.h`
> 核心概念：请求队列、电梯算法、IDE/ATA硬盘驱动

---

## 15.1 块设备驱动架构

```
文件系统 / 内存管理
    │ ll_rw_block(READ/WRITE, buffer_head)
    ▼
通用块设备层 (ll_rw_blk.c)
    │ make_request()  → 创建请求
    │ add_request()   → 加入请求队列（电梯排序）
    │ blk_dev[major].request_fn()  → 触发设备驱动
    ▼
具体驱动 (hd.c / floppy.c / ramdisk.c)
    │ do_hd_request() / do_floppy_request()
    │ 操作硬件（IDE寄存器 / DMA）
    ▼
中断处理
    │ read_intr() / write_intr()（IO完成中断）
    │ 标记 b_uptodate=1, 解锁 b_lock=0
    │ wake_up(b_wait)
    │ 处理下一个请求
    ▼
等待IO的进程被唤醒
```

## 15.2 request 结构和请求队列

```c
// kernel/blk_drv/blk.h
struct request {
    int dev;              // 设备号（-1表示空闲）
    int cmd;              // 命令：READ 或 WRITE
    int errors;           // 错误计数
    unsigned long sector; // 起始扇区号
    unsigned long nr_sectors; // 扇区数（一次总是2，即1KB=1个文件系统块）
    char *buffer;         // 数据缓冲区（= buffer_head.b_data）
    struct task_struct *waiting; // 等待此请求完成的进程
    struct buffer_head *bh;      // 对应的缓冲区头
    struct request *next; // 下一个请求（链表）
};

// 全局请求池
struct request request[NR_REQUEST];  // NR_REQUEST=32

// 块设备描述符
struct blk_dev_struct {
    void (*request_fn)(void);         // 驱动的请求处理函数
    struct request *current_request;  // 当前正在处理的请求
};

struct blk_dev_struct blk_dev[NR_BLK_DEV] = {
    {NULL, NULL},   // 0: no_dev
    {NULL, NULL},   // 1: 内存设备(ramdisk)
    {NULL, NULL},   // 2: 软盘
    {NULL, NULL},   // 3: 硬盘
    ...
};
```

## 15.3 ll_rw_block() —— 块IO入口

```c
// kernel/blk_drv/ll_rw_blk.c 第145~155行
void ll_rw_block(int rw, struct buffer_head *bh)
{
    unsigned int major;

    // 获取主设备号，检查驱动是否存在
    if ((major=MAJOR(bh->b_dev)) >= NR_BLK_DEV ||
        !(blk_dev[major].request_fn)) {
        printk("Trying to read nonexistent block-device\n\r");
        return;
    }
    make_request(major, rw, bh);
}
```

## 15.4 make_request() —— 创建并提交IO请求

```c
// kernel/blk_drv/ll_rw_blk.c 第88~143行
static void make_request(int major, int rw, struct buffer_head *bh)
{
    struct request *req;
    int rw_ahead;

    // READA/WRITEA 是预读/预写，如果缓冲区被锁定则直接放弃
    if (rw_ahead = (rw==READA || rw==WRITEA)) {
        if (bh->b_lock) return;
        if (rw == READA) rw = READ;
        else rw = WRITE;
    }

    lock_buffer(bh);  // 加锁，防止并发

    // 优化：如果数据已满足要求，不需要IO
    if ((rw==WRITE && !bh->b_dirt) || (rw==READ && bh->b_uptodate)) {
        unlock_buffer(bh);
        return;
    }

repeat:
    // 从请求池中找空槽
    // 写请求只能用前2/3（保留1/3给读请求，读操作优先级更高）
    if (rw == READ)
        req = request + NR_REQUEST;
    else
        req = request + ((NR_REQUEST*2)/3);

    while (--req >= request)
        if (req->dev < 0) break;  // dev=-1表示空闲

    if (req < request) {          // 没有空请求槽
        if (rw_ahead) { unlock_buffer(bh); return; }
        sleep_on(&wait_for_request);  // 等待有人释放请求槽
        goto repeat;
    }

    // 填写请求
    req->dev = bh->b_dev;
    req->cmd = rw;
    req->errors = 0;
    req->sector = bh->b_blocknr << 1;   // 1文件系统块 = 2个512字节扇区
    req->nr_sectors = 2;
    req->buffer = bh->b_data;
    req->waiting = NULL;
    req->bh = bh;
    req->next = NULL;

    add_request(major+blk_dev, req);     // 加入请求队列
}
```

## 15.5 add_request() —— 电梯算法排序

```c
// kernel/blk_drv/ll_rw_blk.c 第64~86行
static void add_request(struct blk_dev_struct *dev, struct request *req)
{
    struct request *tmp;

    req->next = NULL;
    cli();  // 关中断，原子操作

    if (req->bh) req->bh->b_dirt = 0;  // 提交写请求前清除脏标志

    if (!(tmp = dev->current_request)) {
        // 队列为空，直接设为当前请求并立即执行
        dev->current_request = req;
        sti();
        (dev->request_fn)();  // 立即调用驱动处理函数
        return;
    }

    // 使用 SCAN（电梯）算法插入排序：
    // 磁头单向移动，按扇区号顺序处理请求，减少磁头移动距离
    for (; tmp->next; tmp=tmp->next)
        if ((IN_ORDER(tmp, req) || !IN_ORDER(tmp, tmp->next)) &&
             IN_ORDER(req, tmp->next))
            break;

    req->next = tmp->next;
    tmp->next = req;
    sti();
}

// IN_ORDER 宏：判断 a 的扇区号 <= b 的扇区号
#define IN_ORDER(s1,s2) \
    ((s1)->cmd<(s2)->cmd || ((s1)->cmd==(s2)->cmd && \
    ((s1)->dev < (s2)->dev || ((s1)->dev == (s2)->dev && \
    (s1)->sector < (s2)->sector))))
```

**电梯算法示意：**
```
磁头当前位置: 扇区 100

请求队列（按扇区排序后）：
  50 → 80 → 120 → 200 → 300

磁头移动：100 → 120 → 200 → 300 → (回头) → 80 → 50
（比随机顺序处理减少了大量磁头移动）
```

## 15.6 硬盘驱动 hd.c

### 硬盘数据结构

```c
// kernel/blk_drv/hd.c
struct hd_i_struct {
    int head, sect, cyl, wpcom, lzone, ctl;
    // head: 磁头数
    // sect: 每道扇区数
    // cyl:  柱面数
    // wpcom: 写预补偿柱面
    // lzone: 落笔柱面
    // ctl:  控制字节
};

// 分区表（每块硬盘最多4个分区）
static struct hd_struct {
    long start_sect;  // 分区起始扇区（相对磁盘开头）
    long nr_sects;    // 分区扇区总数
} hd[5*MAX_HD];
// hd[0] = 第一块硬盘整体
// hd[1..4] = 第一块硬盘的4个分区
// hd[5] = 第二块硬盘整体
// hd[6..9] = 第二块硬盘的4个分区
```

### IDE 端口操作

```c
// include/linux/hdreg.h 中定义的IDE寄存器
// IDE主通道寄存器：
#define HD_DATA     0x1f0  // 数据寄存器（16位，读写扇区数据）
#define HD_ERROR    0x1f1  // 错误寄存器（读）/ 特征寄存器（写）
#define HD_NSECTOR  0x1f2  // 扇区计数
#define HD_SECTOR   0x1f3  // 起始扇区号（LBA 7:0）
#define HD_LCYL     0x1f4  // 柱面低字节 / LBA 15:8
#define HD_HCYL     0x1f5  // 柱面高字节 / LBA 23:16
#define HD_CURRENT  0x1f6  // 驱动器/磁头选择 / LBA 27:24
#define HD_STATUS   0x1f7  // 状态寄存器（读）/ 命令寄存器（写）
#define HD_PRECOMP  HD_ERROR  // 写预补偿（写命令时用）

// 状态寄存器位
#define BUSY_STAT   0x80   // 控制器忙
#define READY_STAT  0x40   // 磁盘就绪
#define DRQ_STAT    0x08   // 数据请求（可以传输数据）
#define ERR_STAT    0x01   // 错误
```

### do_hd_request() —— 处理硬盘请求

```c
// kernel/blk_drv/hd.c
void do_hd_request(void)
{
    int i, r;
    unsigned int block, dev;
    unsigned int sec, head, cyl;
    unsigned int nsect;

    INIT_REQUEST;  // 检查 current_request 是否有效

    dev = MINOR(current_request->dev);  // 分区号
    block = current_request->sector;    // 起始扇区

    // 将逻辑扇区号转换为 CHS (柱面/磁头/扇区)
    block += hd[dev].start_sect;
    dev /= 5;  // 0=第一块盘, 1=第二块盘

    // CHS转换：
    // CHS = (柱面, 磁头, 扇区)
    // 扇区号从1开始（BIOS约定）
    __asm__("divl %4" : "=a" (block), "=d" (sec)
        : "0" (block), "1" (0), "r" (hd_info[dev].sect));
    __asm__("divl %4" : "=a" (cyl), "=d" (head)
        : "0" (block), "1" (0), "r" (hd_info[dev].head));
    sec++;

    nsect = current_request->nr_sectors;

    // 磁盘需要重新校准？
    if (recalibrate) {
        recalibrate = 0;
        hd_out(dev, hd_info[dev].sect, 0, 0, 0,
            WIN_RESTORE, &recal_intr);  // 发送复位命令
        return;
    }

    // 发送读/写命令给磁盘控制器
    if (current_request->cmd == WRITE) {
        hd_out(dev, nsect, sec, head, cyl, WIN_WRITE, &write_intr);
        // 等待DRQ就绪，写入第一个扇区的数据
        for (i=0; i<10000 && !(r=inb_p(HD_STATUS)&DRQ_STAT); i++);
        if (!r) { bad_rw_intr(); do_hd_request(); return; }
        port_write(HD_DATA, current_request->buffer, 256);  // 写256个字=512字节
    } else if (current_request->cmd == READ) {
        hd_out(dev, nsect, sec, head, cyl, WIN_READ, &read_intr);
        // 命令发出，等待中断（磁盘读完后触发IRQ14）
    }
}
```

### 中断驱动的异步IO

```c
// 读中断处理函数（在 _hd_interrupt 中被调用）
static void read_intr(void)
{
    if (win_result()) {  // 检查是否有错误
        bad_rw_intr();
        do_hd_request();
        return;
    }

    // 从数据端口读取一个扇区（256个字=512字节）到缓冲区
    port_read(HD_DATA, current_request->buffer, 256);
    current_request->errors = 0;
    current_request->buffer += 512;
    current_request->sector++;

    if (--current_request->nr_sectors) {
        // 还有更多扇区要读，继续等待下一次中断
        do_hd = &read_intr;  // 设置下次中断的处理函数
        return;
    }

    // 所有扇区读完：唤醒等待进程，处理下一个请求
    end_request(1);   // b_uptodate=1, 解锁, wake_up
    do_hd_request();  // 处理下一个请求
}
```

## 15.7 sys_setup() —— 读取分区表

```c
// kernel/blk_drv/hd.c 第71~145行
int sys_setup(void *BIOS)
{
    static int callable = 1;
    // ...（只能调用一次）

    // 从 setup.s 收集的硬件数据获取硬盘参数
    // 读取0x90080处的hd_info

    // 读取每块硬盘的MBR（第0扇区，512字节）
    // 解析分区表（MBR偏移446处，4个16字节分区项）
    for (drive=0; drive<NR_HD; drive++) {
        if (!(bh = bread(0x300+drive*5, 0))) {
            printk("Unable to read partition table of drive %d\n\r", drive);
            panic("...");
        }
        if (bh->b_data[510] != 0x55 || (unsigned char)bh->b_data[511] != 0xAA) {
            printk("Bad partition table on drive %d\n\r", drive);
            panic("...");
        }
        // 解析4个分区项
        p = 0x1BE + (void *)bh->b_data;  // 分区表位置（MBR偏移446）
        for (i=1; i<5; i++, p++) {
            hd[i+5*drive].start_sect = p->start_sect;
            hd[i+5*drive].nr_sects   = p->nr_sects;
        }
        brelse(bh);
    }

    // 挂载根文件系统
    setup_root_dev();
}
```

## 15.8 blk.h 中的关键宏

```c
// kernel/blk_drv/blk.h

// 当前请求是否有效，无效时返回
#define INIT_REQUEST \
    repeat: \
    if (!current_request) return; \
    if (MAJOR(current_request->dev) != MAJOR_NR) \
        panic("request list destroyed"); \
    if (!current_request->bh) { ... }

// 处理完当前请求：更新缓冲区，唤醒等待进程，移至下一请求
extern inline void end_request(int uptodate)
{
    DEVICE_OFF(current_request->dev);  // 关设备（如软盘马达）
    if (current_request->bh) {
        current_request->bh->b_uptodate = uptodate;
        unlock_buffer(current_request->bh);  // 解锁
    }
    if (!uptodate) {
        printk(DEVICE_NAME " I/O error\n\r");
        printk("dev %04x, block %d\n\r",
            current_request->dev, current_request->bh->b_blocknr);
    }
    wake_up(&current_request->waiting);
    wake_up(&wait_for_request);
    current_request->dev = -1;  // 标记请求槽为空闲
    current_request = current_request->next;  // 移到下一个请求
}
```

## 15.9 总结

| 层次 | 文件 | 职责 |
|------|------|------|
| 通用块层 | ll_rw_blk.c | 请求池管理，电梯排序，分发到驱动 |
| 硬盘驱动 | hd.c | CHS转换，IDE命令，中断处理 |
| 软盘驱动 | floppy.c | DMA传输，马达控制 |
| 内存盘 | ramdisk.c | 直接内存访问，无IO延迟 |

**关键设计：中断驱动的异步IO**
```
- 提交请求后立即返回（不阻塞）
- 等待IO的进程通过 wait_on_buffer 睡眠
- 磁盘完成后触发中断 → 驱动设置结果 → 唤醒进程
- 中间CPU可以运行其他进程（异步、高效）
```
