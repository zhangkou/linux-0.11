## 16.1 字符设备驱动架构

```
用户程序 read("/dev/tty", buf, n)
    │
    ▼ sys_read() → tty_read(channel, buf, nr)
    │
    ▼ TTY层 (tty_io.c)
    ├── 行规程 copy_to_cooked()
    │     ├── 回显处理
    │     ├── 特殊字符处理（^C → SIGINT, ^D → EOF）
    │     └── CR/NL转换
    │
    ├── secondary队列（加工后的输入，供read()读取）
    ├── read_q队列（原始输入，由设备驱动写入）
    └── write_q队列（待输出数据，由设备驱动发出）
    │
    ▼ 具体设备
    ├── console（VGA文本模式 + 键盘）
    │     ├── con_write()   → 向VGA显存写字符
    │     └── keyboard_interrupt() → 键盘扫描码→ASCII
    └── serial（RS-232串口）
          ├── rs_write()    → 触发UART发送中断
          └── rs1/2_interrupt() → 接收中断
```

## 16.2 tty 数据结构

```c
// include/linux/tty.h

// 环形队列：大小必须是2的幂（用 & 代替 % 运算）
struct tty_queue {
    unsigned long data;       // 对串口：端口号；对控制台：未使用
    unsigned long head;       // 写指针（队列头部，新数据写入处）
    unsigned long tail;       // 读指针（队列尾部，消费者读取处）
    struct task_struct *proc_list; // 等待该队列的进程
    char buf[TTY_BUF_SIZE];   // 1024字节环形缓冲区
};

// 关键宏
#define TTY_BUF_SIZE  1024
#define EMPTY(a)  ((a).head == (a).tail)           // 队列为空
#define FULL(a)   (!LEFT(a))                       // 队列已满
#define LEFT(a)   (((a).tail-(a).head-1)&(TTY_BUF_SIZE-1)) // 剩余空间
#define GETCH(queue,c)  (c=(queue).buf[(queue).tail]; INC((queue).tail))
#define PUTCH(c,queue)  ((queue).buf[(queue).head]=(c); INC((queue).head))

// TTY主结构（每个终端设备一个）
struct tty_struct {
    struct termios termios;       // 终端属性（输入/输出/本地标志）
    int pgrp;                     // 前台进程组（接收信号的目标）
    int stopped;                  // 是否暂停输出（^S/^Q）
    void (*write)(struct tty_struct *tty); // 底层写函数指针
    struct tty_queue read_q;      // 原始输入队列（设备→行规程）
    struct tty_queue write_q;     // 输出队列（行规程→设备）
    struct tty_queue secondary;   // 加工后的输入队列（行规程→read()）
};
```

## 16.3 三个队列的数据流

```
  键盘中断             行规程               read()系统调用
  ─────────           ────────             ──────────────
  keyboard_interrupt
      │
      │ 扫描码→ASCII
      ▼
  ┌──────────┐        copy_to_cooked()    ┌────────────────┐
  │  read_q  │ ──────────────────────→   │  secondary     │ ←── tty_read()
  │ (原始输入)│  CR→NL转换               │ (加工后输入)   │     取字符给用户
  └──────────┘  ^C→SIGINT               └────────────────┘
                ^U→清行
                回显→write_q
                        │
                        ▼
                ┌──────────┐
                │  write_q │ ──→ con_write() ──→ VGA显存
                │ (待输出)  │     或 rs_write()  → UART
                └──────────┘
```

## 16.4 copy_to_cooked() —— 行规程核心

```c
// tty_io.c 第145行
void copy_to_cooked(struct tty_struct *tty)
{
    signed char c;

    // 从原始队列取字符，加工后放入secondary队列
    while (!EMPTY(tty->read_q) && !FULL(tty->secondary)) {
        GETCH(tty->read_q, c);

        // ===== 输入字符转换 =====
        if (c == 13) {
            if (I_CRNL(tty)) c = 10;  // CR→NL（默认开启：回车变换行）
        } else if (c == 10 && I_NLCR(tty))
            c = 13;  // NL→CR（反向转换，一般不开）
        if (I_UCLC(tty)) c = tolower(c);  // 大写→小写

        // ===== 规范模式（CANON）下的行编辑 =====
        if (L_CANON(tty)) {
            if (c == KILL_CHAR(tty)) {  // ^U：删除整行
                while (!EMPTY(tty->secondary) &&
                       LAST(tty->secondary) != 10 &&
                       LAST(tty->secondary) != EOF_CHAR(tty)) {
                    // 回显退格（擦除已显示的字符）
                    if (L_ECHO(tty)) { PUTCH(127, tty->write_q); tty->write(tty); }
                    DEC(tty->secondary.head);  // 从secondary删除字符
                }
                continue;
            }
            if (c == ERASE_CHAR(tty)) {  // ^H 或 DEL：退格删字符
                if (!EMPTY(tty->secondary) && LAST(tty->secondary) != 10) {
                    if (L_ECHO(tty)) { PUTCH(127, tty->write_q); tty->write(tty); }
                    DEC(tty->secondary.head);
                }
                continue;
            }
            if (c == STOP_CHAR(tty))  { tty->stopped = 1; continue; }  // ^S
            if (c == START_CHAR(tty)) { tty->stopped = 0; continue; }  // ^Q
        }

        // ===== 信号生成 =====
        if (L_ISIG(tty)) {
            if (c == INTR_CHAR(tty)) {        // ^C → SIGINT
                tty_intr(tty, INTMASK);
                continue;
            }
            if (c == QUIT_CHAR(tty)) {        // ^\ → SIGQUIT
                tty_intr(tty, QUITMASK);
                continue;
            }
        }

        // ===== 回显 =====
        if (L_ECHO(tty)) {
            if (c == 10) {
                PUTCH(10, tty->write_q); PUTCH(13, tty->write_q);  // NL→CRNL
            } else if (c < 32) {
                if (L_ECHOCTL(tty)) {
                    PUTCH('^', tty->write_q);     // 控制字符回显为 ^X
                    PUTCH(c + 64, tty->write_q);
                }
            } else
                PUTCH(c, tty->write_q);
            tty->write(tty);  // 立即刷新到终端
        }

        // 换行或EOF：增加行计数（用于CANON模式判断一行是否完整）
        if (c == 10 || c == EOF_CHAR(tty))
            tty->secondary.data++;

        PUTCH(c, tty->secondary);  // 放入加工后队列
    }
    wake_up(&tty->secondary.proc_list);  // 唤醒等待输入的read()
}
```

**INIT_C_CC 默认控制字符表：**
```
^C (0x03)  VINTR   → SIGINT（中断）
^\  (0x1C) VQUIT   → SIGQUIT（退出+core dump）
DEL(0x7F)  VERASE  → 退格删字符
^U (0x15)  VKILL   → 删除整行
^D (0x04)  VEOF    → 文件结束
^Q (0x11)  VSTART  → 恢复输出（XON）
^S (0x13)  VSTOP   → 暂停输出（XOFF）
^Z (0x1A)  VSUSP   → SIGTSTP（暂停进程）
```

## 16.5 tty_read() / tty_write() —— TTY读写接口

```c
// tty_io.c 第230行
int tty_read(unsigned channel, char *buf, int nr)
{
    struct tty_struct *tty = &tty_table[channel];
    char c, *b = buf;

    while (nr > 0) {
        if (current->signal) break;  // 有待处理信号，中断read

        // 规范模式：没有完整的行（secondary.data == 0），等待
        if (EMPTY(tty->secondary) ||
            (L_CANON(tty) && !tty->secondary.data && LEFT(tty->secondary) > 20)) {
            sleep_if_empty(&tty->secondary);  // 睡眠等待
            continue;
        }

        do {
            GETCH(tty->secondary, c);
            if (c == EOF_CHAR(tty) && L_CANON(tty))
                return (b - buf);   // ^D：返回已读字节数（可能是0）
            put_fs_byte(c, b++);    // 复制到用户空间
            if (!--nr) break;
        } while (nr > 0 && !EMPTY(tty->secondary));

        if (L_CANON(tty))   // 规范模式：读完一行就返回
            if (b - buf) break;
    }
    return (b - buf);
}

// tty_io.c 第290行
int tty_write(unsigned channel, char *buf, int nr)
{
    struct tty_struct *tty = channel + tty_table;
    char c, *b = buf;

    while (nr > 0) {
        sleep_if_full(&tty->write_q);  // write_q满时等待
        if (current->signal) break;

        while (nr > 0 && !FULL(tty->write_q)) {
            c = get_fs_byte(b);  // 从用户空间取字符
            if (O_POST(tty)) {   // 输出后处理
                if (c == '\n' && O_NLCR(tty)) {
                    PUTCH(13, tty->write_q);  // NL→CRNL（先输出CR）
                    continue;
                }
                if (O_LCUC(tty)) c = toupper(c);  // 小写→大写
            }
            b++; nr--;
            PUTCH(c, tty->write_q);
        }
        tty->write(tty);  // 调用底层写函数刷新输出
        if (nr > 0) schedule();  // write_q满了但还有数据，主动调度
    }
    return (b - buf);
}
```

## 16.6 console.c —— VGA 文本模式控制台

### VGA 文本模式内存布局

```
VGA显存（彩色模式）：0xB8000 ~ 0xBFFFF（32KB）
VGA显存（单色模式）：0xB0000 ~ 0xB7FFF

文本模式显存格式（每字符2字节）：
┌──────────────────────────────────┐
│ 字节0：ASCII码                   │
│ 字节1：属性字节                  │
│    bit7:   闪烁                  │
│    bit6-4: 背景色 (0=黑,...,7=白)│
│    bit3:   高亮                  │
│    bit2-0: 前景色 (0=黑,...,7=白)│
└──────────────────────────────────┘

默认属性 0x07 = 黑底白字（正常文本）
高亮属性 0x0F = 黑底亮白字（粗体）
反显属性 0x70 = 白底黑字

例：80列×25行 = 2000字符 = 4000字节
```

### con_init() —— 控制台初始化

```c
// console.c 第617行
void con_init(void)
{
    // 从 setup.s 在 0x90000 处存放的硬件信息读取显示参数
    video_num_columns = ORIG_VIDEO_COLS;   // 列数（通常80）
    video_size_row = video_num_columns * 2; // 每行字节数（160）
    video_num_lines = ORIG_VIDEO_LINES;    // 行数（25）
    video_page = ORIG_VIDEO_PAGE;          // 当前显示页
    video_erase_char = 0x0720;             // 空格+默认属性（用于清屏）

    if (ORIG_VIDEO_MODE == 7) {            // 单色模式（MDA）
        video_mem_start = 0xb0000;
        video_port_reg = 0x3b4;
        video_port_val = 0x3b5;
        if ((ORIG_VIDEO_EGA_BX & 0xff) != 0x10) {
            video_type = VIDEO_TYPE_EGAM;
            video_mem_end = 0xb8000;      // EGA单色：32KB
        } else {
            video_type = VIDEO_TYPE_MDA;
            video_mem_end = 0xb2000;      // MDA：8KB
        }
    } else {                               // 彩色模式（CGA/EGA/VGA）
        video_mem_start = 0xb8000;
        video_port_reg = 0x3d4;
        video_port_val = 0x3d5;
        if ((ORIG_VIDEO_EGA_BX & 0xff) != 0x10) {
            video_type = VIDEO_TYPE_EGAC;
            video_mem_end = 0xbc000;      // EGA彩色：16KB（4页）
        } else {
            video_type = VIDEO_TYPE_CGA;
            video_mem_end = 0xba000;      // CGA：8KB（2页）
        }
    }

    // 初始化滚动参数
    origin = video_mem_start;
    scr_end = video_mem_start + video_num_lines * video_size_row;
    top = 0;
    bottom = video_num_lines;

    gotoxy(ORIG_X, ORIG_Y);  // 恢复启动时的光标位置

    // 安装键盘中断（IRQ1 → 向量 0x21）
    set_trap_gate(0x21, &keyboard_interrupt);
    outb_p(inb_p(0x21) & 0xfd, 0x21);  // 8259A：允许键盘中断
}
```

### con_write() —— VGA 字符输出（含 ANSI 转义序列解析）

```c
// console.c 第445行
void con_write(struct tty_struct *tty)
{
    int nr;
    char c;

    nr = CHARS(tty->write_q);
    while (nr--) {
        GETCH(tty->write_q, c);
        switch (state) {
        case 0:  // 普通字符状态
            if (c > 31 && c < 127) {    // 可打印字符
                // 写入显存：字符+属性 = 2字节
                asm("movb _attr,%%ah; movw %%ax,%1"
                    :: "a"(c), "m"(*(short*)pos) : "ax");
                pos += 2; x++;
            } else if (c == 27) state = 1;  // ESC：进入转义序列
            else if (c == 10 || c == 11 || c == 12) lf();  // LF/VT/FF
            else if (c == 13) cr();    // CR：回到行首
            else if (c == 127) del();  // DEL：退格删除
            else if (c == 8) { if(x){x--; pos-=2;} } // BS：退格
            else if (c == 9) { /* Tab：跳到下一个8的倍数列 */ }
            else if (c == 7) sysbeep(); // BEL：响铃
            break;

        case 1:  // 收到 ESC
            state = 0;
            if (c == '[') state = 2;   // ESC [ → CSI序列
            else if (c == 'M') ri();   // ESC M → 反向换行
            else if (c == 'D') lf();   // ESC D → 换行
            break;

        case 2:  // 收到 ESC [，准备接收参数
            for (npar = 0; npar < NPAR; npar++) par[npar] = 0;
            npar = 0; state = 3;
            // fall through

        case 3:  // 收集数字参数（分号分隔）
            if (c == ';' && npar < NPAR-1) { npar++; break; }
            else if (c >= '0' && c <= '9') { par[npar] = 10*par[npar]+c-'0'; break; }
            else state = 4;  // fall through

        case 4:  // 执行 CSI 命令
            state = 0;
            switch (c) {
                case 'A': gotoxy(x, y - par[0]); break;   // 光标上移
                case 'B': gotoxy(x, y + par[0]); break;   // 光标下移
                case 'C': gotoxy(x + par[0], y); break;   // 光标右移
                case 'D': gotoxy(x - par[0], y); break;   // 光标左移
                case 'H': gotoxy(par[1]-1, par[0]-1); break; // 光标定位
                case 'J': csi_J(par[0]); break;  // 清屏（0/1/2）
                case 'K': csi_K(par[0]); break;  // 清行（0/1/2）
                case 'L': csi_L(par[0]); break;  // 插入行
                case 'M': csi_M(par[0]); break;  // 删除行
                case 'm': csi_m(); break;          // 设置属性（颜色）
                case 'r': /* 设置滚动区域 */ break;
            }
        }
    }
    set_cursor();  // 更新硬件光标位置（写 CRTC 寄存器）
}
```

### EGA/VGA 快速滚屏原理

```c
// console.c 第107行 scrup()
// EGA/VGA 特有的快速滚屏：通过修改显存起始地址实现

static void scrup(void)
{
    if (video_type == VIDEO_TYPE_EGAC || video_type == VIDEO_TYPE_EGAM) {
        if (!top && bottom == video_num_lines) {
            // 全屏滚动：只移动 origin 指针（不复制数据！）
            origin += video_size_row;   // 起始地址+一行
            pos += video_size_row;
            scr_end += video_size_row;
            if (scr_end > video_mem_end) {
                // 显存用完，需要真正复制数据
                // 用 rep movsl 把整屏内容复制到显存开头
                // 然后重置 origin = video_mem_start
            }
            // 最后一行清空
            set_origin();  // 写 CRTC 起始地址寄存器
        }
    }
}
```

```
EGA快速滚屏示意（16KB 显存，25行×160字节=4000字节，可容纳4屏）：

初始：
origin ────→ [第1行][第2行]...[第25行]
                                        ← scr_end

滚动一行后（只移动指针！）：
           [旧第1行(废弃)][第2行]...[第25行][新空行]
origin ──────────────────→               ← scr_end
             CRTC告诉显卡从新origin开始显示
```

## 16.7 keyboard.S —— 键盘中断处理

### 键盘中断流程

```c
// keyboard.S 第37行
_keyboard_interrupt:
    pushl %eax / %ebx / %ecx / %edx / %ds / %es
    movl $0x10, %eax    // 加载内核数据段
    mov %ax, %ds; mov %ax, %es

    xorl %al, %al
    inb $0x60, %al       // 从键盘控制器读取扫描码（端口 0x60）

    cmpb $0xe0, %al; je set_e0   // 0xE0 = 扩展键前缀
    cmpb $0xe1, %al; je set_e1   // 0xE1 = 特殊扩展前缀

    call key_table(,%eax,4)      // 查 key_table 跳转表，处理扫描码

    // 应答键盘控制器（端口 0x61）
    inb $0x61, %al
    orb $0x80, %al; outb %al, $0x61  // 先置位
    andb $0x7F, %al; outb %al, $0x61  // 再清除（脉冲）

    movb $0x20, %al; outb %al, $0x20  // 发 EOI 给 8259A

    pushl $0
    call _do_tty_interrupt    // 调用 copy_to_cooked()
    addl $4, %esp

    pop %es / %ds / %edx / %ecx / %ebx / %eax
    iret
```

### 扫描码处理机制

```
mode 字节（记录修饰键状态）：
  bit0: 左Shift按下
  bit1: 右Shift按下
  bit2: Ctrl按下
  bit3: Ctrl按下（右Ctrl/右Ctrl）
  bit4: Alt按下
  bit5: Alt按下（右Alt）
  bit6: CapsLock有效
  bit7: CapsLock键正在按下（防抖）

leds 字节（LED状态）：
  bit0: Scroll Lock
  bit1: Num Lock（默认开启）
  bit2: Caps Lock

key_table[] 跳转表（256个条目，每条4字节）：
  扫描码 0x01 → do_escape
  扫描码 0x1C → do_enter
  扫描码 0x2A → lshift（左Shift按下）
  扫描码 0xAA → unlshift（左Shift释放）
  ...
```

### put_queue() —— 将ASCII码送入 read_q

```asm
// keyboard.S 第88行
put_queue:
    pushl %ecx; pushl %edx
    movl _table_list, %edx    // 取控制台的 read_q 地址
    movl head(%edx), %ecx     // 取队列头指针

1:  movb %al, buf(%edx,%ecx) // 写入环形缓冲区
    incl %ecx
    andl $size-1, %ecx         // 环绕（size=1024，必须是2的幂）
    cmpl tail(%edx), %ecx
    je 3f                       // 队列满，丢弃
    shrdl $8, %ebx, %eax       // 从 ebx:eax 中移出下一个字节
    je 2f                       // 没有更多字节
    shrl $8, %ebx
    jmp 1b

2:  movl %ecx, head(%edx)     // 更新队列头
    movl proc_list(%edx), %ecx // 唤醒等待者
    testl %ecx, %ecx; je 3f
    movl $0, (%ecx)            // 清除睡眠标志（wake_up效果）
3:  popl %edx; popl %ecx; ret
```

**特殊键处理示例（方向键）：**
```
↑键扫描码: 0x48（或 E0,48 for 增强键盘）
↑键 → ESC [ A（3字节VT100序列）

在 keyboard.S 中：
cur: movb cur_table(%eax), %al    // 'A'（上）
     shll $16, %eax               // 保存
     movw $0x5b1b, %ax            // 低16位 = ESC '['
     xorl %ebx, %ebx
     jmp put_queue                // 将 ESC [ A 送入 read_q
```

## 16.8 serial.c —— 串口（UART）驱动

### UART 寄存器（以 COM1: 0x3F8 为基址）

```
端口          名称           作用
──────────────────────────────────────────
0x3F8 (读)   接收数据寄存器  读取接收到的字节
0x3F8 (写)   发送数据寄存器  写入待发送的字节
0x3F9         中断使能寄存器  控制哪些事件产生中断
0x3FA         中断ID寄存器    查询中断原因
0x3FB         线控寄存器      数据位/停止位/校验设置
              bit7=DLAB：置1时 0x3F8/9 变为波特率因子
0x3FC         调制控制寄存器  DTR/RTS/OUT2控制
0x3FD         线状态寄存器    接收/发送就绪等状态
```

### rs_init() —— 串口初始化

```c
// serial.c 第37行
void rs_init(void)
{
    // 安装中断处理程序
    set_intr_gate(0x24, rs1_interrupt);  // IRQ4 → COM1
    set_intr_gate(0x23, rs2_interrupt);  // IRQ3 → COM2

    // 初始化两个串口
    init(tty_table[1].read_q.data);   // COM1: 0x3F8
    init(tty_table[2].read_q.data);   // COM2: 0x2F8

    // 8259A：允许 IRQ3 和 IRQ4（COM1/COM2）
    outb(inb_p(0x21) & 0xE7, 0x21);
}

// 初始化单个串口（设置 2400 波特率，8N1）
static void init(int port)
{
    outb_p(0x80, port+3);  // DLAB=1，准备设置波特率
    outb_p(0x30, port);    // 波特率因子低字节：48 → 115200/48 = 2400bps
    outb_p(0x00, port+1);  // 波特率因子高字节
    outb_p(0x03, port+3);  // DLAB=0，8位数据，1停止位，无校验（8N1）
    outb_p(0x0b, port+4);  // DTR+RTS+OUT2（允许中断输出到CPU）
    outb_p(0x0d, port+1);  // 使能中断：接收就绪、发送空、线路状态、调制变化
    (void)inb(port);       // 读清接收寄存器
}
```

### rs_write() —— 触发串口发送

```c
// serial.c 第53行
void rs_write(struct tty_struct *tty)
{
    cli();
    // write_q 不空时，使能发送保持寄存器空中断（bit1）
    // UART 发送完当前字节后会产生中断，中断处理程序继续发下一个字节
    if (!EMPTY(tty->write_q))
        outb(inb_p(tty->write_q.data+1) | 0x02, tty->write_q.data+1);
    sti();
}
// 注意：rs_write 只是"启动"发送，实际字节由 rs_io.s 中的中断程序逐字节发送
```

**串口发送的中断驱动流程：**
```
tty_write() 向 write_q 写入数据
    │
    ▼ rs_write() 使能 UART 发送中断
    │
    │ UART 发送完当前字节 → 产生中断
    ▼
rs1_interrupt (rs_io.s)
    │ 从 write_q 取一字节
    │ outb 到 UART 数据寄存器
    │ 如果 write_q 空了：禁用发送中断
    ▼
重复直到 write_q 清空
```

## 16.9 tty_io 与系统调用的连接

```c
// fs/char_dev.c（字符设备的 read/write 分发）
int rw_char(int rw, int dev, char *buf, int count, off_t *pos)
{
    crw_ptr call_addr;

    if (MAJOR(dev) > MAX_CHRDEV) return -ENODEV;
    if (!(call_addr = crw_table[MAJOR(dev)])) return -ENODEV;
    return call_addr(rw, MINOR(dev), buf, count, pos);
}

// TTY 字符设备（主设备号 4/5）
static crw_ptr crw_table[] = {
    NULL,          // 0: 无设备
    NULL,          // 1: 内存设备
    NULL,          // 2: 软盘（块设备）
    NULL,          // 3: 硬盘（块设备）
    rw_ttyx,       // 4: /dev/ttyX
    rw_tty,        // 5: /dev/tty（当前终端）
    ...
};

static int rw_ttyx(int rw, unsigned minor, char *buf, int count, off_t *pos)
{
    return ((rw==READ) ?
        tty_read(minor, buf, count) :
        tty_write(minor, buf, count));
}
```

## 16.10 总结

**字符设备驱动层次：**

| 层次 | 文件 | 职责 |
|------|------|------|
| 系统调用层 | `fs/char_dev.c` | 根据设备号分发到 tty_read/tty_write |
| TTY公共层 | `tty_io.c` | 行规程、三队列管理、信号发送 |
| 控制台驱动 | `console.c` | VGA显存直写、ANSI转义解析、滚屏 |
| 键盘驱动 | `keyboard.S` | 扫描码→ASCII、修饰键处理 |
| 串口驱动 | `serial.c` + `rs_io.s` | UART寄存器操作、中断驱动收发 |

**关键设计思想：**
```
1. 三队列设计（read_q / write_q / secondary）
   - 分离"原始输入"和"加工后输入"
   - 行规程处理在两者之间（中断安全）

2. 中断驱动的异步IO
   - 键盘/串口接收：中断→填充 read_q→唤醒进程
   - 串口发送：启动中断→逐字节从 write_q 发送
   - 控制台写：同步直接写显存（无需等待）

3. 规范模式 vs 原始模式
   - 规范模式（ICANON）：按行处理，支持行编辑（^U/^H/^C）
   - 原始模式（!ICANON）：逐字节传递，适用于vi等程序

4. EGA/VGA 快速滚屏
   - 修改显存起始寄存器，而非复制整屏数据
   - 只在显存用完时才真正进行内存复制
```
