# 第10章：信号处理

> 源码文件：`kernel/signal.c`、`kernel/exit.c`
> 核心概念：信号位图、信号投递、用户态信号处理函数的调用机制

---

## 10.1 Linux 0.11 的信号系统概述

Linux 0.11 实现了基本的 Unix 信号机制：
- 32种信号，用位图表示（task_struct.signal 是一个 long）
- 信号的发送：设置对应位
- 信号的处理：在系统调用/定时器中断返回前检查
- 用户自定义处理函数：通过修改用户栈来"欺骗"CPU执行

## 10.2 信号定义

```c
// include/signal.h（部分）
#define SIGHUP   1   // 挂断（终端关闭）
#define SIGINT   2   // 中断（Ctrl+C）
#define SIGQUIT  3   // 退出（Ctrl+\）
#define SIGILL   4   // 非法指令
#define SIGTRAP  5   // 断点陷阱
#define SIGABRT  6   // 异常终止
#define SIGFPE   8   // 浮点异常
#define SIGKILL  9   // 强制终止（不可捕获）
#define SIGSEGV  11  // 段错误（访问非法地址）
#define SIGPIPE  13  // 管道破裂
#define SIGALRM  14  // 定时器信号
#define SIGTERM  15  // 终止（可捕获）
#define SIGCHLD  17  // 子进程状态改变
#define SIGCONT  18  // 继续执行
#define SIGSTOP  19  // 暂停（不可捕获）

// task_struct 中的信号字段：
// signal: long  (32位位图，bit(n-1) = 信号n是否待处理)
// blocked: long (32位位图，bit(n-1) = 信号n是否被屏蔽)
```

## 10.3 sigaction 结构

```c
// include/signal.h
struct sigaction {
    void (*sa_handler)(int);  // 处理函数：NULL=默认，1=忽略，其他=用户函数
    sigset_t sa_mask;          // 处理时额外屏蔽的信号位图
    int sa_flags;              // 标志
    void (*sa_restorer)(void); // 恢复函数（从信号处理返回后调用）
};

// sa_flags 标志
#define SA_ONESHOT  0x01  // 处理一次后恢复默认
#define SA_NOMASK   0x02  // 处理时不自动屏蔽本信号
```

## 10.4 发送信号

```c
// kernel/exit.c 第35~44行（内部函数）
static inline int send_sig(long sig, struct task_struct *p, int priv)
{
    if (!p || sig<1 || sig>32)
        return -EINVAL;
    // 权限检查：有特权(priv)，或同一euid，或是超级用户
    if (priv || (current->euid == p->euid) || suser())
        p->signal |= (1 << (sig-1));  // 设置信号位
    else
        return -EPERM;
    return 0;
}
```

```c
// kernel/exit.c 第60~81行
int sys_kill(int pid, int sig)  // kill() 系统调用
{
    struct task_struct **p = NR_TASKS + task;
    int err, retval = 0;

    if (!pid)          // pid=0: 发给同进程组所有进程
        while (--p > &FIRST_TASK) {
            if (*p && (*p)->pgrp == current->pid)
                if (err=send_sig(sig,*p,1)) retval = err;
        }
    else if (pid > 0)  // pid>0: 发给指定PID进程
        while (--p > &FIRST_TASK) {
            if (*p && (*p)->pid == pid)
                if (err=send_sig(sig,*p,0)) retval = err;
        }
    else if (pid == -1) // pid=-1: 发给所有进程
        while (--p > &FIRST_TASK)
            if (err = send_sig(sig,*p,0)) retval = err;
    else               // pid<-1: 发给进程组|pid|
        while (--p > &FIRST_TASK)
            if (*p && (*p)->pgrp == -pid)
                if (err = send_sig(sig,*p,0)) retval = err;
    return retval;
}
```

## 10.5 信号检测时机

信号检测在每次**从内核返回用户态时**进行：

```asm
; kernel/system_call.s ret_from_sys_call 中
; （每次系统调用或时钟中断返回时都会执行这段代码）

ret_from_sys_call:
    movl _current, %eax
    cmpl _task, %eax           ; 跳过进程0
    je 3f
    cmpw $0x0f, CS(%esp)       ; 必须是返回到用户态
    jne 3f
    cmpw $0x17, OLDSS(%esp)
    jne 3f

    movl signal(%eax), %ebx    ; 取信号位图
    movl blocked(%eax), %ecx
    notl %ecx                   ; 取反 = 未屏蔽信号掩码
    andl %ebx, %ecx             ; 找未屏蔽的待处理信号
    bsfl %ecx, %ecx             ; 找最低位（最小信号号-1）
    je 3f                       ; 没有信号则返回
    btrl %ecx, %ebx             ; 清除该信号位
    movl %ebx, signal(%eax)
    incl %ecx                   ; 信号号 = 位号+1
    pushl %ecx
    call _do_signal              ; 处理信号！
    popl %eax
3:
    ; 正常返回用户态
```

## 10.6 do_signal() —— 信号投递的精髓

这是信号处理中最复杂的函数，它通过**修改用户栈**实现信号处理函数的调用：

```c
// kernel/signal.c 第82~119行
void do_signal(long signr, long eax, long ebx, long ecx, long edx,
    long fs, long es, long ds,
    long eip, long cs, long eflags,
    unsigned long *esp, long ss)
{
    unsigned long sa_handler;
    long old_eip = eip;
    struct sigaction *sa = current->sigaction + signr - 1;
    int longs;
    unsigned long *tmp_esp;

    sa_handler = (unsigned long) sa->sa_handler;

    // 情况1：忽略信号（sa_handler == 1）
    if (sa_handler == 1)
        return;

    // 情况2：默认处理
    if (!sa_handler) {
        if (signr == SIGCHLD)
            return;  // SIGCHLD默认忽略
        else
            do_exit(1<<(signr-1));  // 其他信号默认终止进程
    }

    // 情况3：用户自定义处理函数
    if (sa->sa_flags & SA_ONESHOT)
        sa->sa_handler = NULL;  // 一次性信号，处理后恢复默认

    // ★★关键：修改内核栈上的 EIP，让iret后执行信号处理函数
    *(&eip) = sa_handler;

    // 在用户栈上构造信号处理函数的调用帧
    longs = (sa->sa_flags & SA_NOMASK) ? 7 : 8;
    *(&esp) -= longs;          // 扩展用户栈
    verify_area(esp, longs*4); // 验证用户栈地址合法

    tmp_esp = esp;
    // 从高到低依次压入（用户栈从高地址向低地址增长）：
    put_fs_long((long) sa->sa_restorer, tmp_esp++); // sa_restorer（返回地址）
    put_fs_long(signr, tmp_esp++);                   // 信号号（处理函数的参数）
    if (!(sa->sa_flags & SA_NOMASK))
        put_fs_long(current->blocked, tmp_esp++);    // 保存当前信号屏蔽字
    put_fs_long(eax, tmp_esp++);   // 保存被中断时的寄存器
    put_fs_long(ecx, tmp_esp++);
    put_fs_long(edx, tmp_esp++);
    put_fs_long(eflags, tmp_esp++);
    put_fs_long(old_eip, tmp_esp++); // 原来的EIP（信号处理后恢复点）

    current->blocked |= sa->sa_mask;  // 临时屏蔽额外的信号
}
```

**信号处理的"虚假调用"原理：**
```
内核栈（iret前）：      用户栈（do_signal修改后）：
EIP = sa_handler  ←    用户栈顶：sa_restorer
                         +4:  signr
                         +8:  blocked(旧)
                        +12:  eax(旧)
                        +16:  ecx(旧)
                        +20:  edx(旧)
                        +24:  eflags(旧)
                        +28:  old_eip（信号处理完后返回的地址）

iret 执行后：
1. CPU恢复EIP = sa_handler → 开始执行信号处理函数
2. 信号处理函数认为自己是被"正常调用"的
3. 函数参数（signr）已在用户栈上
4. 函数执行 ret 时，弹出 sa_restorer → 进入恢复函数
5. sa_restorer 恢复寄存器（eax/ecx/edx/eflags），恢复信号屏蔽字
6. sa_restorer 执行 ret，弹出 old_eip → 回到被中断的位置
```

## 10.7 信号处理流程图

```
进程正常运行
    │
    │ (收到信号，如 Ctrl+C → SIGINT)
    │ p->signal |= (1<<(SIGINT-1))
    │
    ▼
时钟中断或系统调用返回
    │
    ▼
ret_from_sys_call: 检测到信号
    │
    │ do_signal(SIGINT, ...)
    │   修改内核栈上的EIP = 用户信号处理函数地址
    │   在用户栈上构造假调用帧
    │
    ▼
iret 返回用户态
    │ EIP = 用户信号处理函数
    ▼
用户信号处理函数执行
    │
    │ ret
    ▼
sa_restorer 执行
    │ 恢复寄存器和信号屏蔽字
    │ ret
    ▼
回到被中断的原始代码
```

## 10.8 sys_signal() 与 sys_sigaction()

```c
// kernel/signal.c 第48~61行
int sys_signal(int signum, long handler, long restorer)
{
    struct sigaction tmp;

    if (signum<1 || signum>32 || signum==SIGKILL)
        return -1;  // SIGKILL 不可捕获

    tmp.sa_handler = (void (*)(int)) handler;
    tmp.sa_mask = 0;
    tmp.sa_flags = SA_ONESHOT | SA_NOMASK;  // signal()是一次性的
    tmp.sa_restorer = (void (*)(void)) restorer;

    // 返回旧的处理函数（signal()的返回值）
    handler = (long) current->sigaction[signum-1].sa_handler;
    current->sigaction[signum-1] = tmp;
    return handler;
}
```

**signal() 和 sigaction() 的区别：**
```
signal():    SA_ONESHOT = 处理一次后恢复默认（不可靠信号）
sigaction(): 持久注册，可以精确控制 sa_mask/sa_flags
```

## 10.9 不可捕获/不可屏蔽的信号

```c
// kernel/sched.c
#define _S(nr) (1<<((nr)-1))
#define _BLOCKABLE (~(_S(SIGKILL) | _S(SIGSTOP)))
// _BLOCKABLE: 除SIGKILL和SIGSTOP外都可屏蔽
```

| 信号 | 特性 |
|------|------|
| SIGKILL (9) | 不可捕获、不可忽略、不可屏蔽 |
| SIGSTOP (19) | 不可捕获、不可忽略、不可屏蔽 |
| SIGCHLD (17) | 默认忽略（不退出进程） |
| 其他 | 可以用 signal()/sigaction() 自定义 |

## 10.10 总结

| 机制 | 实现 |
|------|------|
| 信号存储 | `task_struct.signal`（32位位图） |
| 屏蔽信号 | `task_struct.blocked`（位图） |
| 发送信号 | `p->signal |= (1<<(sig-1))` |
| 检测时机 | 系统调用/时钟中断返回用户态前 |
| 投递机制 | 修改内核栈EIP + 在用户栈构造假帧 |
| 恢复机制 | sa_restorer 恢复寄存器，ret回原代码 |
