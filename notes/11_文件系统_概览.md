# 第11章：文件系统概览（Minix FS）

> 源码文件：`fs/super.c`、`fs/inode.c`、`fs/bitmap.c`、`include/linux/fs.h`
> 核心概念：Minix文件系统结构、超级块、inode、目录项

---

## 11.1 Minix 文件系统布局

Linux 0.11 使用 **Minix 文件系统**（由 Andrew Tanenbaum 设计），磁盘布局如下：

```
磁盘（或分区）
┌────────────────────────────────────────────────────────┐
│  引导块         │ 块0（保留，含分区引导代码，或全零）       │ 1块
├────────────────────────────────────────────────────────┤
│  超级块         │ 块1（文件系统元数据）                    │ 1块
├────────────────────────────────────────────────────────┤
│  inode位图      │ 块2开始（标记哪些inode已使用）            │ s_imap_blocks块
├────────────────────────────────────────────────────────┤
│  数据块位图     │ 紧接其后（标记哪些数据块已使用）           │ s_zmap_blocks块
├────────────────────────────────────────────────────────┤
│  inode表        │ 每个inode 32字节，存储文件元数据         │ (s_ninodes/32+1)块
├────────────────────────────────────────────────────────┤
│  数据块区       │ 实际文件内容（目录也是数据块）             │ 剩余所有块
└────────────────────────────────────────────────────────┘

块大小 = 1024字节（BLOCK_SIZE）
```

## 11.2 超级块（super_block）

```c
// include/linux/fs.h
struct d_super_block {       // 磁盘上的超级块（读入后使用此结构）
    unsigned short s_ninodes;    // inode总数
    unsigned short s_nzones;     // 数据区块总数（逻辑块）
    unsigned short s_imap_blocks; // inode位图占用块数
    unsigned short s_zmap_blocks; // 数据块位图占用块数
    unsigned short s_firstdatazone; // 第一个数据块号
    unsigned short s_log_zone_size;  // log2(块/逻辑块), 通常=0(1块/逻辑块)
    unsigned long s_max_size;    // 最大文件大小（字节）
    unsigned short s_magic;      // 魔数: 0x137F (Minix FS v1)
};

// 内存中的超级块（比磁盘版多了几个字段）
struct super_block {
    unsigned short s_ninodes;
    unsigned short s_nzones;
    unsigned short s_imap_blocks;
    unsigned short s_zmap_blocks;
    unsigned short s_firstdatazone;
    unsigned short s_log_zone_size;
    unsigned long s_max_size;
    unsigned short s_magic;
    /* 以下是内存中才有的字段 */
    struct buffer_head * s_imap[8]; // inode位图缓冲区指针（最多8块）
    struct buffer_head * s_zmap[8]; // 数据块位图缓冲区指针（最多8块）
    unsigned short s_dev;           // 设备号
    struct m_inode * s_isup;        // 根目录inode（挂载点）
    struct m_inode * s_imount;      // 挂载到哪个inode上
    unsigned long s_time;           // 最后修改时间
    struct task_struct * s_wait;    // 等待超级块的进程
    unsigned char s_lock;           // 超级块锁
    unsigned char s_rd_only;        // 只读标志
    unsigned char s_dirt;           // 已修改（脏）
};
```

## 11.3 inode（索引节点）

每个文件/目录对应一个inode，存储文件元数据和数据块索引：

```c
// include/linux/fs.h
struct d_inode {               // 磁盘inode（32字节）
    unsigned short i_mode;     // 文件类型和权限（rwxrwxrwx）
    unsigned short i_uid;      // 属主用户ID
    unsigned long  i_size;     // 文件大小（字节）
    unsigned long  i_time;     // 最后修改时间
    unsigned char  i_gid;      // 属主组ID
    unsigned char  i_nlinks;   // 硬链接数
    unsigned short i_zone[9];  // 块地址数组
};

struct m_inode {               // 内存inode（包含更多运行时信息）
    unsigned short i_mode;
    unsigned short i_uid;
    unsigned long  i_size;
    unsigned long  i_mtime;
    unsigned char  i_gid;
    unsigned char  i_nlinks;
    unsigned short i_zone[9];  // 数据块索引（见下方说明）
    /* 内存额外字段 */
    struct task_struct * i_wait;
    unsigned long  i_atime;    // 访问时间
    unsigned long  i_ctime;    // 创建时间
    unsigned short i_dev;      // 所在设备
    unsigned short i_num;      // inode号
    unsigned short i_count;    // 引用计数
    unsigned char  i_lock;
    unsigned char  i_dirt;
    unsigned char  i_pipe;     // 是否是管道
    unsigned char  i_mount;    // 是否是挂载点
    unsigned char  i_seek;
    unsigned char  i_update;
};
```

**i_zone[9] 数据块索引结构：**
```
i_zone[0..6]  : 直接块（7块 × 1024B = 7KB）
i_zone[7]     : 一级间接块（一个块存512个块号 → 512KB）
i_zone[8]     : 二级间接块（512 × 512个块号 → 256MB，但实际受s_max_size限制）

最大文件大小 = 7×1KB + 512×1KB + 512×512×1KB = 7KB + 512KB + 256MB
但 d_super_block.s_max_size 通常设为更小的值（约64MB）
```

**i_mode 权限位：**
```
bit15-12: 文件类型
  0100 = 普通文件
  0010 = 字符设备
  0060 = 块设备
  0040 = 目录
  0120 = 符号链接
  0010 = 管道

bit11-9:  setuid/setgid/sticky
bit8-6:   属主权限 (rwx)
bit5-3:   属组权限 (rwx)
bit2-0:   其他权限 (rwx)
```

## 11.4 目录项（dir_entry）

```c
// include/linux/fs.h
#define NAME_LEN 14         // 文件名最大长度
#define ROOT_INO 1          // 根目录的inode号永远是1

struct dir_entry {
    unsigned short inode;   // inode号（0=空目录项）
    char name[NAME_LEN];    // 文件名（不含路径）
};
// 目录项大小 = 2 + 14 = 16字节
// 一个1KB数据块可存 1024/16 = 64 个目录项
```

**目录本质是一个特殊的文件：**
```
目录文件的数据块内容示例（/usr/ 目录）：
偏移0:  {inode=1,  name="."}      // 当前目录
偏移16: {inode=5,  name=".."}     // 父目录
偏移32: {inode=12, name="bin"}
偏移48: {inode=20, name="lib"}
偏移64: {inode=0,  name=""}       // 空槽（已删除的条目）
...
```

## 11.5 super.c —— 超级块管理

```c
// fs/super.c（关键函数）

// 全局超级块缓存（最多同时挂载8个文件系统）
struct super_block super_block[NR_SUPER];

// 读取超级块
static struct super_block *read_super(int dev)
{
    struct super_block *s;
    struct buffer_head *bh;
    int i, block;

    // 检查是否已在缓存中
    if (s = get_super(dev)) return s;

    // 找空槽
    for (s = 0+super_block ;; s++) {
        if (s >= NR_SUPER+super_block) return NULL;
        if (!s->s_dev) break;
    }

    s->s_dev = dev;
    s->s_isup = NULL;
    s->s_imount = NULL;
    s->s_time = 0;
    s->s_rd_only = 0;
    s->s_dirt = 0;
    lock_super(s);

    // 读块1（超级块数据）
    if (!(bh = bread(dev,1))) {
        s->s_dev = 0;
        free_super(s);
        return NULL;
    }

    // 复制磁盘超级块到内存超级块
    *((struct d_super_block *)s) = *((struct d_super_block *)bh->b_data);
    brelse(bh);

    // 验证魔数
    if (s->s_magic != SUPER_MAGIC) {
        s->s_dev = 0;
        free_super(s);
        return NULL;
    }

    // 读取inode位图（s_imap_blocks块）
    for (i=0; i<I_MAP_SLOTS; i++) s->s_imap[i] = NULL;
    for (i=0; i<Z_MAP_SLOTS; i++) s->s_zmap[i] = NULL;
    block = 2;  // 从块2开始（块0=引导，块1=超级块）
    for (i=0; i<s->s_imap_blocks && i<I_MAP_SLOTS; i++)
        if (s->s_imap[i]=bread(dev,block)) block++;
        else break;

    // 读取数据块位图
    for (i=0; i<s->s_zmap_blocks && i<Z_MAP_SLOTS; i++)
        if (s->s_zmap[i]=bread(dev,block)) block++;
        else break;

    // 确保位图0号不被分配（inode从1开始，块从s_firstdatazone开始）
    s->s_imap[0]->b_data[0] |= 1;
    s->s_zmap[0]->b_data[0] |= 1;

    free_super(s);
    return s;
}
```

## 11.6 inode.c —— inode 管理

```c
// fs/inode.c（关键函数）

// 内存inode缓存
struct m_inode inode_table[NR_INODE] = {{0,},};  // NR_INODE=32

// 读取inode（从缓存或磁盘）
struct m_inode *iget(int dev, int nr)
{
    struct m_inode *inode, *empty;

    // 先在缓存中查找
    inode = inode_table;
    while (inode < NR_INODE+inode_table) {
        if (inode->i_dev == dev && inode->i_num == nr) {
            // 在缓存中找到
            wait_on_inode(inode);
            if (inode->i_dev == dev && inode->i_num == nr) {
                inode->i_count++;
                return inode;

        inode++;

    // 读取该块，提取inode数据
    ...
}

// 释放inode（减引用计数，引用为0时写回磁盘）
void iput(struct m_inode *inode)
{
    if (!inode) return;
    wait_on_inode(inode);
    if (!inode->i_count)
        panic("iput: trying to free free inode");

    if (inode->i_pipe) {
        wake_up(&inode->i_wait);
        if (--inode->i_count) return;
        // 释放管道页
        free_page(inode->i_size);
        inode->i_count = 0;
        inode->i_dirt = 0;
        inode->i_pipe = 0;
        return;
    }

    if (!inode->i_dev) { inode->i_count--; return; }

    if (!inode->i_nlinks) {
        // 硬链接数为0：真正删除文件（释放所有数据块）
        truncate(inode);
        free_inode(inode);
        return;
    }

    if (inode->i_dirt)
        write_inode(inode);  // 写脏数据回磁盘

    inode->i_count--;
}
```

## 11.7 bitmap.c —— 位图分配

```c
// fs/bitmap.c
// 在位图中分配一个空闲项（inode或数据块）
static int find_first_zero(void *addr)  // 汇编实现，找第一个为0的位
{
    int res;
    __asm__("cld\n"
        "1:\tlodsl\n\t"       // 从内存加载一个long
        "notl %%eax\n\t"      // 取反：原来是0的位变成1
        "bsfl %%eax,%%edx\n\t" // 从低位找第一个1（即原来的0）
        "je 2f\n\t"
        "addl %%edx,%%ecx\n\t" // 计算位偏移
        "jmp 3f\n"
        "2:\taddl $32,%%ecx\n\t" // 这个long全为1，继续下一个
        "cmpl $8192,%%ecx\n\t"   // 超过8192位（=8块×1024字节×8位）？
        "jl 1b\n"
        "3:"
        :"=c" (res):"c" (0),"S" (addr):"ax","dx","si");
    return res;
}

// 分配新inode
int new_inode(int dev)
{
    struct m_inode *inode;
    struct super_block *sb;
    struct buffer_head *bh;
    int i, j;

    // 在位图中找第一个空闲位
    if (!(sb = get_super(dev)) || !(inode = get_empty_inode()))
        return 0;

    j = 8192;
    for (i=0; i<8; i++)
        if (bh = sb->s_imap[i])
            if ((j=find_first_zero(bh->b_data)) < 8192)
                break;

    if (!bh || j >= 8192 || j+i*8192 > sb->s_ninodes) {
        iput(inode);
        return 0;
    }

    // 设置位图中该位
    if (set_bit(j, bh->b_data)) panic("new_inode: bit already set");
    bh->b_dirt = 1;   // 标记位图块已修改

    inode->i_count = 1;
    inode->i_nlinks = 1;
    inode->i_dev = dev;
    inode->i_uid = current->euid;
    inode->i_gid = current->egid;
    inode->i_dirt = 1;
    inode->i_num = j + i*8192;
    inode->i_mtime = inode->i_atime = inode->i_ctime = CURRENT_TIME;
    return inode->i_num;
}
```

## 11.8 文件系统的层次结构

```
用户程序
    │ open/read/write/close 等系统调用
    ▼
系统调用层 (fs/open.c, fs/read_write.c)
    │
    ▼
VFS层（Linux 0.11没有真正的VFS，但函数结构已体现此思想）
    │
    ▼
Minix文件系统层
    ├── namei.c    路径解析（字符串→inode）
    ├── inode.c    inode管理
    ├── super.c    超级块管理
    ├── bitmap.c   空闲块分配
    └── buffer.c   块缓存（文件系统与磁盘之间的缓冲层）
    │
    ▼
块设备层 (ll_rw_blk.c)
    │
    ▼
磁盘驱动 (hd.c / floppy.c)
```

## 11.9 总结

| 结构 | 大小 | 作用 |
|------|------|------|
| 超级块 | 32字节（磁盘） | 文件系统全局元数据 |
| inode（磁盘） | 32字节 | 文件元数据 + 9个块指针 |
| inode（内存） | 约100字节 | 磁盘inode + 运行时信息 |
| 目录项 | 16字节 | 文件名(14B) + inode号(2B) |
| 数据块 | 1024字节 | 文件内容或目录列表 |
| 内存inode缓存 | 32个 | NR_INODE=32 |
| 超级块缓存 | 8个 | NR_SUPER=8（最多挂载8个FS） |
